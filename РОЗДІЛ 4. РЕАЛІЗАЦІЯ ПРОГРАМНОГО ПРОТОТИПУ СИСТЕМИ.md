# РОЗДІЛ 4. РЕАЛІЗАЦІЯ ПРОГРАМНОГО ПРОТОТИПУ СИСТЕМИ

У даному розділі описано практичну реалізацію розроблених алгоритмів у вигляді функціонуючого програмного прототипу GenAI-системи. Прототип складається із серверної частини на базі Node.js, що виконує роль оркестратора та ізольованого середовища (Sandbox), і клієнтського SPA-додатка (Single Page Application) на базі Vue.js та D3.js для інтерактивного рендерингу візуалізацій.

## 4.1 Реалізація модуля визначення схеми даних (Schema Inference)

Модуль Schema Inference є першим етапом серверного pipeline та реалізує алгоритм, формалізований у підрозділі 3.1. Його завданням є автоматичний аналіз завантаженого CSV-файлу та формування компактного JSON-артефакту схеми, який передається до всіх наступних модулів системи. Підхід концептуально відповідає ідеї Hierarchical Data Context (HDC), запропонованій у роботі TiInsight [1, 2], де схема витягується з великих структур даних до їх передачі до мовної моделі.

Реалізацію модуля побудовано на базі бібліотеки `papaparse` для парсингу CSV та власної евристики класифікації типів стовпців. Програмний код модуля наведено у лістингу 4.1.

**Лістинг 4.1 – Програмний код модуля Schema Inference**

```javascript
const Papa = require('papaparse');

/**
 * Визначає тип стовпця за ієрархією евристичних правил (підрозділ 3.1).
 * Класифікація виконується детерміновано, без звернення до LLM [14, 15].
 */
function inferColumnType(values) {
    const nonEmpty = values.filter(v => v !== null && v !== '');
    if (nonEmpty.length === 0) return 'Categorical';

    const total = nonEmpty.length;

    // Перевірка Temporal: ISO 8601 або dd/mm/yyyy (поріг 80%)
    const temporalCount = nonEmpty.filter(v =>
        /^\d{4}-\d{2}-\d{2}/.test(v) || /^\d{2}[./-]\d{2}[./-]\d{4}$/.test(v)
    ).length;
    if (temporalCount / total >= 0.8) return 'Temporal';

    // Перевірка Numeric: парсяться як дійсне число (поріг 90%)
    const numericCount = nonEmpty.filter(v => !isNaN(parseFloat(v))).length;
    if (numericCount / total >= 0.9) return 'Numeric';

    return 'Categorical';
}

/**
 * Обчислює статистичні характеристики залежно від типу стовпця.
 */
function computeStats(values, type) {
    const nonEmpty = values.filter(v => v !== null && v !== '');
    const nullRate = parseFloat(((values.length - nonEmpty.length) / values.length).toFixed(3));

    if (type === 'Numeric') {
        const nums = nonEmpty.map(Number);
        const mean = nums.reduce((a, b) => a + b, 0) / nums.length;
        return {
            min: Math.min(...nums),
            max: Math.max(...nums),
            mean: parseFloat(mean.toFixed(2)),
            null_rate: nullRate
        };
    }
    if (type === 'Temporal') {
        const sorted = [...nonEmpty].sort();
        return { range: [sorted[0], sorted[sorted.length - 1]], null_rate: nullRate };
    }
    // Categorical
    const freq = {};
    nonEmpty.forEach(v => { freq[v] = (freq[v] || 0) + 1; });
    const mode = Object.keys(freq).reduce((a, b) => freq[a] > freq[b] ? a : b);
    return { cardinality: new Set(nonEmpty).size, mode, null_rate: nullRate };
}

/**
 * Головна функція модуля. Повертає Schema JSON (Лістинг 3.1).
 */
function inferSchema(csvText) {
    const parsed = Papa.parse(csvText, { header: true, skipEmptyLines: true });
    const rows = parsed.data;
    const columns = parsed.meta.fields;
    const totalRows = rows.length;

    // Аналіз перших N=100 рядків для ефективності
    const sampleRows = rows.slice(0, 100);

    const columnDescriptions = columns.map(col => {
        const values = sampleRows.map(row => row[col] ?? null);
        const type = inferColumnType(values);
        const stats = computeStats(values, type);
        return { name: col, type, ...stats };
    });

    // Стратифікована вибірка 3 рядків для промпту
    const sample = rows
        .sort(() => Math.random() - 0.5)
        .slice(0, 3);

    return { total_rows: totalRows, columns: columnDescriptions, sample };
}
```

Результатом роботи модуля є JSON-об’єкт розміром ~200–400 токенів незалежно від обсягу вхідного CSV, що відповідає вимозі приватності (підрозділ 2.3.3) — до LLM передається лише структурний опис, а не повний масив даних. Детерміновану класифікацію типів стовпців обрано з огляду на схильність LLM до одновимірної інтерпретації табличних структур [14, 15].

## 4.2 Реалізація модуля вибору типу візуалізації

Модуль вибору типу візуалізації реалізує гібридний алгоритм, формалізований у підрозділі 3.2: rule-based логіка обробляє типові комбінації стовпців детерміновано, тоді як неоднозначні випадки делегуються мовній моделі. Такий підхід концептуально відповідає алгоритму TiChart [1, 2] та забезпечує нульову вартість інференсу для переважної більшості запитів [35].

Програмний код модуля наведено у лістингу 4.2.

**Лістинг 4.2 – Програмний код модуля вибору типу візуалізації**

```javascript
/**
 * Підраховує кількість стовпців кожного типу зі схеми.
 */
function countColumnTypes(schema) {
    return schema.columns.reduce((acc, col) => {
        acc[col.type] = (acc[col.type] || 0) + 1;
        return acc;
    }, { Temporal: 0, Numeric: 0, Categorical: 0 });
}

/**
 * Визначає кардинальність першого категоріального стовпця.
 */
function getCategoricalCardinality(schema) {
    const col = schema.columns.find(c => c.type === 'Categorical');
    return col ? col.cardinality : Infinity;
}

/**
 * Rule-based вибір типу графіка за таблицею пріоритетів (Таблиця 3.1).
 * Повертає null, якщо жодне правило не спрацювало → делегування LLM.
 */
function applyRules(counts, schema) {
    const { Temporal: nT, Numeric: nN, Categorical: nC } = counts;
    const kappa = getCategoricalCardinality(schema);
    const KAPPA_LOW = 6;

    if (nT === 1 && nN >= 2) return 'multiline';
    if (nT === 1 && nN === 1) return 'line';
    if (nC === 2 && nN === 1) return 'grouped-bar';
    if (nC === 1 && nN === 1) return 'bar';
    if (nC === 1 && nN === 2) return 'scatter-color';
    if (nN === 2)             return 'scatter';
    if (nC === 1 && nN === 0 && kappa <= KAPPA_LOW) return 'pie';

    return null; // неоднозначний випадок
}

/**
 * Делегування вибору LLM для неоднозначних випадків.
 */
async function selectChartTypeLLM(schema, openai) {
    const candidates = ['bar', 'line', 'scatter', 'pie', 'heatmap', 'multiline'];
    const prompt = `Given this dataset schema:\n${JSON.stringify(schema.columns)}\n` +
                   `Choose the most appropriate chart type from: ${candidates.join(', ')}.\n` +
                   `Respond ONLY with valid JSON: {"chartType": "...", "reasoning": "..."}`;

    const response = await openai.chat.completions.create({
        model: 'gpt-4o',
        messages: [{ role: 'user', content: prompt }],
        temperature: 0
    });

    const raw = response.choices[0].message.content;
    const parsed = JSON.parse(raw.replace(/```json|```/g, '').trim());
    return { chartType: parsed.chartType, source: 'llm', reasoning: parsed.reasoning };
}

/**
 * Головна функція модуля.
 * Повертає кортеж (chartType, source, reasoning) для логування в Розділі 5.
 */
async function selectChartType(schema, openai) {
    const counts = countColumnTypes(schema);
    const ruleResult = applyRules(counts, schema);

    if (ruleResult) {
        return {
            chartType: ruleResult,
            source: 'rule-based',
            reasoning: `Matched rule: nT=${counts.Temporal}, nN=${counts.Numeric}, nC=${counts.Categorical}`
        };
    }

    // Неоднозначний випадок — делегування LLM
    return await selectChartTypeLLM(schema, openai);
}
```

Поле `source` у вихідному кортежі фіксує, яким саме шляхом було прийнято рішення — rule-based або LLM. Це значення логується оркестратором та використовується в експериментальному аналізі (Розділ 5) для оцінки частки випадків, що потребують звернення до мовної моделі. Результат `chartType` передається до оркестратора (підрозділ 4.4) як жорстке обмеження в системному промпті, що унеможливлює вибір LLM альтернативного типу графіка на етапі генерації коду [14, 16].

## 4.3 Реалізація серверної частини та дворівневої валідації коду

Серверну частину розроблюваної системи реалізовано на базі середовища виконання Node.js із використанням вебфреймворку Express.js. Її головним завданням є оркестрація процесів згідно з розробленими алгоритмами: виведення схеми (Schema Inference), формування блокових промптів та валідація D3.js-коду, згенерованого великою мовною моделлю.

Відповідно до алгоритму валідації (підрозділ 3.4), для гарантування безпеки та функціональної валідності згенерованого коду реалізовано дворівневий механізм перевірки. Необхідність такої перевірки підтверджується дослідженням LEVER [20, 21], де показано, що виконання згенерованого коду та верифікація результату суттєво покращують метрики успішності генерації. Перший рівень (Static Analysis) здійснює лексичний аналіз згенерованого тексту для виявлення заборонених функцій. Другий рівень (Sandbox Execution) використовує вбудований модуль Node.js `vm` для виконання коду в ізольованому віртуальному контексті з обмеженням часу та пам’яті.

Програмну реалізацію модуля дворівневої валідації наведено у лістингу 4.3.

**Лістинг 4.3 – Програмний код модуля дворівневої валідації та ізольованого виконання**

```javascript
const vm = require('vm');

/**
 * Рівень 1: Статичний аналіз коду (Static Analysis)
 * Перевіряє наявність заборонених API без виконання коду.
 */
function performStaticAnalysis(code) {
    const forbiddenPatterns = [
        'eval(', 'Function(', 'fetch(', 'XMLHttpRequest', 'document.cookie',
        'd3.nest(', 'd3.csv(', 'd3.tsv('
    ];
    for (let pattern of forbiddenPatterns) {
        if (code.includes(pattern)) {
            return {
                isValid: false,
                errorTrace: `Security Error: Usage of '${pattern}' is forbidden.`
            };
        }
    }
    // Перевірка наявності обов'язкової сигнатури функції
    if (!code.includes('function renderChart') && !code.includes('renderChart =')) {
        return { isValid: false, errorTrace: 'Structural Error: renderChart function not found.' };
    }
    return { isValid: true };
}

/**
 * Рівень 2: Виконання в пісочниці (Sandbox Execution)
 * Концептуально відповідає підходу execution feedback [20, 21].
 */
function validateCodeInSandbox(generatedCode, dataSample) {
    const staticCheck = performStaticAnalysis(generatedCode);
    if (!staticCheck.isValid) return staticCheck;

    // Створення ізольованого контексту з об'єктами-заглушками (MockD3)
    const sandbox = {
        data: dataSample,
        console: { log: () => {}, error: () => {} },
        d3: {
            select: () => ({
                append: () => ({
                    attr: () => ({ style: () => ({}), text: () => ({}) })
                }),
                selectAll: () => ({
                    data: () => ({ join: () => ({ attr: () => ({}) }) })
                })
            }),
            scaleLinear: () => ({ domain: () => ({ range: () => ({}) }) }),
            scaleBand:   () => ({ domain: () => ({ range: () => ({ padding: () => ({}) }) }) }),
            scaleTime:   () => ({ domain: () => ({ range: () => ({}) }) }),
            axisBottom: () => ({}), axisLeft: () => ({}),
            max: () => 100, min: () => 0, extent: () => [0, 100]
        },
        document: { createElement: () => ({}), querySelector: () => ({}) }
    };

    const context = vm.createContext(sandbox);

    try {
        const script = new vm.Script(generatedCode);
        // Обмеження часу виконання для захисту від нескінченних циклів
        script.runInContext(context, { timeout: 2000 });
        return { isValid: true, errorTrace: null };
    } catch (error) {
        // Перехоплення stack trace для механізму Self-Refine [22, 23]
        return { isValid: false, errorTrace: error.toString() };
    }
}
```

Реалізований механізм виконує код у режимі суворого обмеження часу (`timeout: 2000` мс), що запобігає зависанню сервера. Якщо на будь-якому з двох рівнів виникає помилка, її текст (`errorTrace`) повертається до оркестратора для ініціації циклу ітеративного виправлення відповідно до методу Self-Refine [24].

## 4.4 Інтеграція з API мовних моделей та конструювання промптів

Для взаємодії з мовними моделями оркестратор використовує динамічне конструювання промптів, що складається з п’яти незалежних блоків відповідно до формули, наведеної у підрозділі 3.3, та патерну Visualization Generator [27, 28, 29]. Це дозволяє гнучко перемикати режими генерації (Zero-shot, Few-shot, CoT [30]) без зміни загальної архітектури.

Взаємодія з моделями реалізована за допомогою офіційної бібліотеки `openai`. Програмну логіку оркестратора наведено у лістингу 4.4.

**Лістинг 4.4 – Програмний код оркестратора генерації D3.js коду**

```javascript
const { OpenAI } = require('openai');
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

/**
 * Основна функція генерації та ітеративного вдосконалення коду.
 * Реалізує цикл Self-Refine [24] з лімітом у MAX_RETRIES ітерацій.
 */
async function generateChartCode(schema, sample, chartInfo, mode = 'zero-shot') {
    const MAX_RETRIES = 3;
    let attempt = 0;
    let validationLog = [];

    // Конструювання системного промпту з 5 блоків (підрозділ 3.3)
    const B_role = "You are an expert D3.js v7 developer. Think step by step.\n";
    const B_schema = `[Schema]\n${JSON.stringify(schema)}\n[Sample]\n${JSON.stringify(sample)}\n`;
    const B_chart = `[Chart Requirement]\nGenerate a: ${chartInfo.chartType}\n` +
                    `X-axis: ${chartInfo.xColumn}, Y-axis: ${chartInfo.yColumn}\n`;
    const B_constraints = "[Constraints]\nReturn ONLY valid JavaScript inside ```javascript." +
                          " Do not use d3.nest(), d3.csv(), d3.tsv().\n";
    const B_shots = mode === 'few-shot' ? getFewShotExample(chartInfo.chartType) : "";

    const systemPrompt = B_role + B_schema + B_chart + B_shots + B_constraints;

    let messages = [
        { role: "system", content: systemPrompt },
        { role: "user", content: "Generate the D3.js visualization code based on the schema." }
    ];

    while (attempt < MAX_RETRIES) {
        const response = await openai.chat.completions.create({
            model: "gpt-4o",
            messages: messages,
            temperature: 0.1 // Низька температура для детермінованості результатів
        });

        // Екстракція коду за допомогою регулярного виразу
        const rawContent = response.choices[0].message.content;
        const codeMatch = rawContent.match(/```javascript([\s\S]*?)```/);
        const generatedCode = codeMatch ? codeMatch[1].trim() : rawContent.trim();

        // Дворівнева валідація (підрозділ 4.3)
        const validationResult = validateCodeInSandbox(generatedCode, sample);

        if (validationResult.isValid) {
            validationLog.push(`Attempt ${attempt + 1}: Success`);
            return {
                status: "success",
                code: generatedCode,
                iterations: attempt + 1,
                chart_type: chartInfo.chartType,
                validation_log: validationLog
            };
        } else {
            // Формування Execution Feedback для Self-Refine [22, 23, 24]
            validationLog.push(`Attempt ${attempt + 1}: Failed - ${validationResult.errorTrace}`);
            messages.push({ role: "assistant", content: rawContent });
            messages.push({
                role: "user",
                content: `Execution failed with error:\n${validationResult.errorTrace}\n` +
                         `Fix the code. Preserve chart type (${chartInfo.chartType}). ` +
                         `Return ONLY the corrected \`\`\`javascript block.`
            });
            attempt++;
        }
    }

    // Fallback-механізм у разі вичерпання ліміту спроб
    return {
        status: "fallback",
        code: getFallbackTemplate(chartInfo.chartType),
        iterations: MAX_RETRIES,
        chart_type: chartInfo.chartType,
        validation_log: validationLog
    };
}
```

У наведеному коді реалізовано механізм екстракції цільового скрипта та формування вихідного артефакту (`status`, `code`, `iterations`, `chart_type`, `validation_log`), структура якого відповідає специфікації підрозділу 3.4. У разі вичерпання ліміту спроб система фіксує статус `"fallback"` та повертає статичний шаблон візуалізації для збереження працездатності інтерфейсу.

## 4.5 Реалізація клієнтської частини системи на базі Vue.js та D3.js

Клієнтську частину системи реалізовано у вигляді односторінкового вебдодатка (SPA) з використанням фреймворку Vue.js (версія 3, Composition API) та бібліотеки візуалізації D3.js (версія 7). Відповідно до розробленої архітектури передачі даних Schema + Sample (підрозділ 2.3.2), повний масив даних залишається виключно в пам’яті клієнтського браузера та не передається до зовнішніх сервісів.

Після успішної генерації та валідації коду на сервері клієнтський додаток отримує текстовий рядок із JavaScript-кодом. Для інтеграції цього коду в DOM-дерево Vue-компонента реалізовано механізм динамічного монтування, концептуально близький до підходу прив’язки даних до DOM-елементів, описаного в системі InterChat [5, 6]. Оскільки отриманий код вже пройшов етап перевірки на безпеку (Static Analysis), його виконання здійснюється через конструктор `Function`, що забезпечує ізоляцію області видимості.

Алгоритм динамічного рендерингу та прив’язки повного набору даних наведено у лістингу 4.5.

**Лістинг 4.5 – Програмний код компонента динамічного рендерингу на Vue.js**

```javascript
import { ref } from 'vue';
import * as d3 from 'd3';

// Реактивне посилання на DOM-контейнер для графіка
const chartContainer = ref(null);

/**
 * Функція динамічного виконання та рендерингу D3.js коду.
 * @param {Object} serverResponse - Відповідь сервера (status, code, chart_type).
 * @param {Array<Object>} fullDataset - Повний масив даних (Full CSV).
 */
function mountGeneratedChart(serverResponse, fullDataset) {
    if (!chartContainer.value) return;

    // Очищення контейнера від попередніх візуалізацій
    d3.select(chartContainer.value).selectAll('*').remove();

    if (serverResponse.status !== 'success' && serverResponse.status !== 'fallback') {
        console.error("Генерація графіка не вдалася.");
        return;
    }

    try {
        const generatedCode = serverResponse.code;

        // Інкапсуляція згенерованого коду у нову функцію.
        // Залежності (d3, data, container) передаються як аргументи —
        // принцип інверсії управління для ізоляції scope [5, 6]
        const executeChart = new Function('d3', 'data', 'container', `
            ${generatedCode}
            if (typeof renderChart === 'function') {
                renderChart(data, container);
            } else {
                throw new Error("Функція renderChart не знайдена у згенерованому коді.");
            }
        `);

        // Виконання з передачею реального об'єкта D3 та повного масиву даних
        executeChart(d3, fullDataset, chartContainer.value);

    } catch (error) {
        console.error("Помилка клієнтського рендерингу:", error);
    }
}
```

У клієнтському коді застосовано принцип інверсії управління: згенерований скрипт не імпортує D3.js самостійно, а отримує її екземпляр (`d3`), повний масив об’єктів (`data`) та посилання на HTML-елемент (`container`) через параметри динамічно створеної функції. Це гарантує, що скрипт матиме доступ лише до виділеного вузла DOM-дерева та дозволяє безперервно оновлювати робочу область при ітеративному візуальному аналізі без перезавантаження сторінки.

## 4.6 Висновки до розділу

У даному розділі описано практичну реалізацію розроблених алгоритмів автоматичної генерації візуалізацій. Модуль Schema Inference реалізує детерміновану класифікацію стовпців за евристичними правилами та формує компактний JSON-артефакт схеми без звернення до LLM [14, 15]. Модуль вибору типу візуалізації реалізує гібридний алгоритм із семи rule-based правил та делегування LLM для неоднозначних випадків [1, 2], що забезпечує детермінованість для переважної більшості запитів [35]. Серверну частину реалізовано з модулями статичного аналізу та виконання в ізольованій пісочниці (`vm`) відповідно до підходу верифікації коду через виконання [20, 21]. Оркестратор конструює п’ятиблоковий промпт згідно з патерном Visualization Generator [27, 28, 29] та забезпечує виконання циклу Self-Refine [24] при взаємодії з API OpenAI. Клієнтський компонент на базі Vue.js динамічно монтує отриманий D3.js-код у DOM-дерево та зв’язує його з повним масивом даних користувача, реалізуючи принцип локалізації даних (підрозділ 2.3.3). Реалізований прототип відповідає архітектурним та функціональним вимогам, що створює основу для переходу до етапу експериментального дослідження (Розділ 5).
